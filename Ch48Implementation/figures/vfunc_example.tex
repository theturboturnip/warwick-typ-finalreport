        \begin{cppcode}
class Animal {
public:
    virtual void talk() = 0;

    virtual ~Animal() = default;
};

class Dog : public Animal {
public:
    void talk() override {
        printf("bark\n");
    }

    ~Dog() override = default;
};

int main() {
    // Virtual call
    Animal* animal = new Dog();
    animal->talk(); // Prints 'bark'
    delete animal;

    Dog dog = {};
    dog.talk();

    return 0;
}
        \end{cppcode}

mov     rdi, rax

# Calling the function statically (dog.talk())
